FROM Водители

	AND Место прописки = 'еее'


--2
SELECT ФИО водителя, № паспорта
FROM Диспетчерская
LEFT JOIN Водители 
ON Диспетчерская.код_водителя = Водители.код_водителя
WHERE Время отбытия IS NULL

SELECT ФИО водителя, № паспорта
FROM Диспетчерская
LEFT JOIN Водители 
ON Диспетчерская.код_водителя = Водители.код_водителя
WHERE Путевка IS NULL


 --3
SELECT рег.номер_авто, время_прибытия, время_отбытия, пункт_назначения
FROM Автомбили авто
JOIN Диспетчерская д
ON авто.код_авто = д.код_авто
JOIN Маршрут м
ON д.путевка = м.код_маршрута
WHERE год_выпуска_авто = 2015
